 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public static ListNode getIntersectionNode(ListNode listNode1, ListNode listNode2) {
        if(listNode1 == null || listNode2 == null) {
            return null;
        }

        ListNode loop1 = findFirstInputLoopNode(listNode1);
        ListNode loop2 = findFirstInputLoopNode(listNode2);

        // ä¸¤ä¸ªæ— ç¯é“¾è¡¨ç›¸äº¤
        if(loop1 == null && loop2 == null) {
            return notLoopNodeIntersect(listNode1, listNode2);
        }

        // ä¸¤ä¸ªé“¾è¡¨éƒ½æœ‰ç¯
        if(loop1 != null && loop2 != null) {
            return loopNodeIntersect(listNode1, loop1, listNode2, loop2);
        }

        // ä¸¤ä¸ªé“¾è¡¨å…¶ä¸­ä¸€ä¸ªæœ‰ç¯ï¼Œä¸å¯èƒ½ç›¸äº¤
        return null;
    }

    /**
     * ä¸¤ä¸ªæœ‰ç¯å•é“¾è¡¨
     * - loop1å’Œloop2çš„åœ°å€ç›¸ç­‰ï¼Œå°±æ˜¯ä¸¤ä¸ªæ— ç¯é“¾è¡¨æ±‚ç¬¬ä¸€ä¸ªç›¸äº¤èŠ‚ç‚¹
     * - loop1å’Œloop2çš„åœ°å€ä¸ç›¸ç­‰ï¼Œéå†é“¾è¡¨ä¸€å¦‚æœåœ¨é“¾è¡¨ä¸­æ²¡é‡åˆ°loop2åˆ™æ— ç›¸äº¤èŠ‚ç‚¹ï¼Œé‡åˆ°loop2åˆ™å­˜åœ¨ç›¸äº¤èŠ‚ç‚¹è¿”å›loop1æˆ–loop2éƒ½å¯ä»¥ã€‚
     */
    public static ListNode loopNodeIntersect(ListNode listNodeOne, ListNode loopOne, ListNode listNodeTwo, ListNode loopTwo) {
        ListNode currentOne = null;
        ListNode currentTwo = null;
        if(loopOne == loopTwo){
            currentOne = listNodeOne;
            currentTwo = listNodeTwo;
            int i = 0;
            while (currentOne != loopOne) {
                currentOne = currentOne.next;
                i++;
            }

            while (currentTwo != loopTwo) {
                currentTwo = currentTwo.next;
                i--;
            }

            // å°†é•¿çš„èµ‹å€¼ç»™lastNodeOne, çŸ­çš„èµ‹å€¼ç»™lastNodeTwo
            currentOne = i > 0 ? listNodeOne : listNodeTwo;
            currentTwo = currentOne == listNodeOne ? listNodeTwo : listNodeOne;

            i = Math.abs(i);
            while (i != 0) {
                currentOne = currentOne.next;
                i--;
            }
            while (currentOne != currentTwo) {
                currentOne = currentOne.next;
                currentTwo = currentTwo.next;
            }
            return currentOne;
        }

        // loop1å’Œloop2çš„åœ°å€ä¸ç›¸ç­‰ï¼Œéå†é“¾è¡¨ä¸€å¦‚æœåœ¨é“¾è¡¨ä¸­æ²¡é‡åˆ°loop2åˆ™æ— ç›¸äº¤èŠ‚ç‚¹ï¼Œé‡åˆ°loop2åˆ™å­˜åœ¨ç›¸äº¤èŠ‚ç‚¹è¿”å›loop1æˆ–loop2éƒ½å¯ä»¥ã€‚
        else {
            currentOne = listNodeOne.next;
            while (currentOne != loopOne) {
                if(currentOne == loopTwo) {
                    return loopTwo;
                }
                currentOne = currentOne.next;
            }
            return null;
        }
    }


    /**
     * ä¸¤ä¸ªæ— ç¯å•é“¾è¡¨
     *
     * éå†ä¸¤ä¸ªé“¾è¡¨å¦‚æœæœ€åä¸€ä¸ªèŠ‚ç‚¹çš„åœ°å€ç›¸åŒåˆ™ç›¸äº¤ã€‚
     * é•¿çš„é“¾è¡¨å…ˆèµ° ä¸¤ä¸ªèŠ‚ç‚¹çš„é•¿åº¦å·® ååœ¨åˆ¤æ–­ä¸¤ä¸ªèŠ‚ç‚¹åœ°å€æ˜¯å¦ç›¸åŒï¼Œç¬¬ä¸€ä¸ªç›¸ç­‰çš„èŠ‚ç‚¹å°±æ˜¯ç›¸äº¤èŠ‚ç‚¹ã€‚
     */
    public static ListNode notLoopNodeIntersect(ListNode listNodeOne, ListNode listNodeTwo) {
        ListNode lastListNodeOne = listNodeOne;
        int nodeLength = 0;
        while (lastListNodeOne.next != null) {
            lastListNodeOne = lastListNodeOne.next;
            nodeLength++;
        }

        ListNode lastListNodeTwo = listNodeTwo;
        while (lastListNodeTwo.next != null) {
            lastListNodeTwo = lastListNodeTwo.next;
            nodeLength--;
        }
        // æœ€åä¸€ä¸ªèŠ‚ç‚¹çš„åœ°å€ä¸ç›¸åŒåˆ™ä¸ä¼šç›¸äº¤
        if(lastListNodeOne != lastListNodeTwo) {
            return null;
        }

        // å°†é•¿çš„èµ‹å€¼ç»™lastNodeOne, çŸ­çš„èµ‹å€¼ç»™lastNodeTwo
        lastListNodeOne = nodeLength > 0 ? listNodeOne : listNodeTwo;
        lastListNodeTwo = lastListNodeOne == listNodeOne ? listNodeTwo : listNodeOne;

        // é•¿çš„èŠ‚ç‚¹å…ˆèµ°æ‰å·®å€¼
        int abs = Math.abs(nodeLength);
        while (abs != 0) {
            lastListNodeOne = lastListNodeOne.next;
            abs--;
        }

        // èµ°åˆ°ç›¸ç­‰ä¸ºæ­¢
        while (lastListNodeOne != lastListNodeTwo) {
            lastListNodeOne = lastListNodeOne.next;
            lastListNodeTwo = lastListNodeTwo.next;
        }
        return lastListNodeOne;
    }

    /**
     * è·å–ç¬¬ä¸€ä¸ªå…¥ç¯èŠ‚ç‚¹ï¼Œæ²¡æœ‰å…¥ç¯èŠ‚ç‚¹åˆ™è¿”å›null
     *
     * å®ç°æ–¹å¼ï¼šåˆ©ç”¨å¿«æ…¢æŒ‡é’ˆ(å¿«2æ…¢1)ï¼Œèµ°åˆ°ç›¸äº¤èŠ‚ç‚¹æ—¶å¿«æŒ‡é’ˆå›åˆ°å¤´ï¼Œä¹‹åå¿«æŒ‡é’ˆæ­¥é•¿æ”¹ä¸º1ï¼Œ
     * é‡æ–°ç›¸é‡åçš„ç¬¬ä¸€ä¸ªèŠ‚ç‚¹å°±æ˜¯å…¥ç¯çš„ç¬¬ä¸€ä¸ªçš„èŠ‚ç‚¹ï¼Œå¿«æŒ‡é’ˆé‡åˆ°nullåˆ™é“¾è¡¨èŠ‚ç‚¹æ— å…¥ç¯èŠ‚ç‚¹
     */
    public static ListNode findFirstInputLoopNode(ListNode head) {
        if(head == null || head.next == null || head.next.next == null) {
            return null;
        }
        ListNode fast = head.next.next, slow = head.next;
        boolean isIntersect = false, firstHead = true;
        while (fast != null) {
            if(fast == slow) {
                // ç¬¬äºŒæ¬¡ç›¸é‡çš„èŠ‚ç‚¹ç›´æ¥è¿”å›
                if(isIntersect) {
                    return slow;
                }
                isIntersect = true;
            }
            slow = slow.next;
            if(fast.next == null) {
                return null;
            }
            if(isIntersect) {
                if(firstHead) {
                    firstHead = false;
                    fast = head;
                }
                fast = fast.next;
            } else {
                fast = fast.next.next;
            }
        }
        return null;
    }
}

    public static Node level(Node node, Node level) {
        if(node == null || level == null) {
            return node;
        }
        Node current = node;
        // åªåˆ é™¤å•é“¾è¡¨æ— ç¯
        while (true) {
            if(current == level) {
                current.val = current.next.val;
                current.next =  current.next.next;
                break;
            } else if(current.next.next == null) {
                current.next = null;
                break;
            }
            current = current.next;
        }
        return node;
    }
