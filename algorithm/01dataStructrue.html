                                                                  class LinkedQueueNode {

    private Node head;

    private Node tail;

    private Integer size = 0;

    public void push(String name) {
        Node node = new Node(name);
        if(this.head == null) {
           this.head = node;
        }
        if(this.tail != null) {
            this.tail.setNext(node);
        }
        this.tail = node;
        this.size++;
    }

    public Node pop() {
        Node head = this.head;
        this.head = this.head.getNext();
        head.setNext(null);
        return head;
    }
}

                                                                  class LinkedStack{

    private Node node;

    public void push(String name) {
        Node node = new Node(name);
        if(this.node == null){
            this.node = node;
            return;
        }

        node.setNext(this.node);
        this.node = node;
    }

    public Node pop() {
        if(this.node == null) {
            return null;
        }
        Node next = this.node;

        this.node = next.getNext();
        return next;
    }
}
