                            // æ— æ•ˆå€¼
                            dp[i + 1][w.peekLast()] == Integer.MAX_VALUE ||
                            // æœ‰æ•ˆåˆ™åŠ ä¸Šè¡¥å¿å€¼åå¤§äºç­‰äºåé¢çš„å€¼
                            dp[i + 1][w.peekLast()] + compensate(w.peekLast(), r, coins[i]) >= dp[i + 1][r]
                            )) {
                        w.pollLast();
                    }
                    w.addLast(r);

                    // è¿‡æœŸçš„æ•°æ®å·¦è¾¹çš„å€¼å¾€åèµ°
                    int overdue = r - coins[i] * (zhangs[i] + 1);
                    if(overdue == w.peekFirst()) {
                        w.pollFirst();
                    }
                    // èµ‹å€¼
                    dp[i][r] = dp[i + 1][w.peekFirst()] + compensate(w.peekFirst(), r, coins[i]);
                }
            }
        }

        return dp[0][aim];
    }

    public static int compensate(int pre, int cur, int coin) {
        return (cur - pre) / coin;
    }

    private static Info getInfo(int[] arr) {
        // ç”¨mapå­˜æ”¾æ•°é‡ä»¥åŠé¢å€¼
        HashMap<Integer, Integer> counts = new HashMap<>();
        for (int value : arr) {
            if (!counts.containsKey(value)) {
                counts.put(value, 1);
            } else {
                counts.put(value, counts.get(value) + 1);
            }
        }
        // éå†mapå°†é¢å€¼å’Œæ•°é‡æ‹†åˆ†æ•°ç»„é‡Œé¢
        int N = counts.size();
        int[] coins = new int[N];
        int[] zhangs = new int[N];
        int index = 0;
        for (Map.Entry<Integer, Integer> entry : counts.entrySet()) {
            coins[index] = entry.getKey();
            zhangs[index++] = entry.getValue();
        }
        return new Info(coins, zhangs);
    }

        private int[] coins;
        private int[] zhangs;

            this.coins = coins;
            this.zhangs = zhangs;
        }
