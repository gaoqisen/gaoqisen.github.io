                                                                          public static int getMinLight(String str) {
        if(str == null || str.isEmpty()) {
            return 0;
        }

        char[] chars = str.toCharArray();
        int light = 0;
        for (int i = 0; i < chars.length; i++) {
            if(chars[i] == 'X') {
                continue;
            } else {
                light++;
                if(i + 1 == chars.length) {
                    break;
                }
                if (chars[i + 1] == 'X') {
                    i += 1;
                } else {
                    i += 2;
                }
            }
        }
        return light;
    }
